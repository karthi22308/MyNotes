#include <iostream>
#include <unordered_map>
using namespace std;

int main()
{
     unordered_map<int, string> map;
     
     //insert operation
     
     map[1]="karthick";
     map.insert({2,"test"});
     
     //iterating 
     for(auto it : map){
         cout<<it.first<<" "<<it.second<<endl;
     }
    //accesisng at particular index;
    cout<<map[5]<<endl<<"jj";
   // cout<<map.at(6)<<endl;
    
    //check if it exists
    
    if(map.find(1)==map.end()){
        cout<<"key 10 not found";
    }
    else{
        cout<<map[1];
    }
    
    return 0;
}
